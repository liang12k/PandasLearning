C == ctrl
space == spacebar

# quit
C-x C-c

# save
C-x C-s

# position cursor by mouse click; similar to iTerm2
alt + <mouse click at position>

# top of file
ESC <
# bottom of file
ESC >

# beginning of line
C-a

# end of line
C-e

# move cursor by skipping by word
# skip fwd
alt + f
# skip bkwd
alt + b

# cut: 
# : set the end cut pt (C-space)
# : position cursor at start pt
# : cut (C-w)
C-space C-w

# paste
C-y
